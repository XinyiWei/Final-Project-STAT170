def error_rate(prediction,observation):
    err_count = 0
    for i in range(len(observation)):
        if prediction[i] != observation[i]:
            err_count += 1
            #print(prediction[i], observation[i])
    return err_count/len(prediction)

def predict(validation):
    result = []
    for i in range(len(validation)):
        posterior = validation[i][1]
        if(posterior< 0.5):
            result.append(0)
        else:
            result.append(1)
    return result
